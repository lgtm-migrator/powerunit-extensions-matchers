020package ch.powerunit.extensions.matchers;
021
022import java.lang.annotation.Documented;
023import java.lang.annotation.ElementType;
024import java.lang.annotation.Inherited;
025import java.lang.annotation.Repeatable;
026import java.lang.annotation.Retention;
027import java.lang.annotation.RetentionPolicy;
028import java.lang.annotation.Target;
029
030/**
031 * This annotation may be use to add DSL syntax for this field to the generated
032 * matcher.
033 * <p>
034 * The main purpose of this annotation is to allow developer to add, simple,
035 * DSL, to the generated one.
036 * <p>
037 * Let's assume you have a field annotated in the following way (included in a
038 * class named {@code PojoWithOverrideField}) :
039 * 
040 * <pre>
041 * &#64;AddToMatcher(suffix = "Suffix", argument = "", body = "// Does nothing, just for sample")
042 * &#64;AddToMatcher(suffix = "SuffixTest", argument = "java.lang.String test", body = "// Does nothing, just for sample")
043 * public String field;
044 * </pre>
045 * 
046 * Then additional DSL method will be added :
047 * 
048 * <pre>
049 * &#64;Override
050 * public PojoWithOverrideFieldMatcher&gt;_PARENT&lt; fieldSuffix() {
051 *      // Does nothing, just for sample
052 *      return this;
053 * }
054 * 
055 * &#64;Override
056 * public PojoWithOverrideFieldMatcher&gt;_PARENT&lt; fieldSuffixTest(java.lang.String test) {
057 *      // Does nothing, just for sample
058 *      return this;
059 * }
060 * </pre>
061 * 
062 * <p>
063 * <i>The annotation must be used on the public getter or the field (it this one
064 * is public). Having a public field and a public getter should be avoided as in
065 * this case it is not sure that this annotation will be detected.</i>
066 * 
067 * @author borettim
068 * @since 0.0.7
069 */
070@Documented
071@Retention(RetentionPolicy.SOURCE)
072@Target({ ElementType.FIELD, ElementType.METHOD })
073@Inherited
074@Repeatable(AddToMatchers.class)
075public @interface AddToMatcher {
076        /**
077         * Specify a suffix to be used after the name generated by the framework.
078         * <p>
079         * The name used inside the DSL will be composed of the fieldName (starting with
080         * a lower case), followed by the value of this suffix. Use an empty String when
081         * no suffix is required.
082         * 
083         * @return the suffix to be used
084         */
085        String suffix();
086
087        /**
088         * Specify the argument to be used.
089         * 
090         * @return the argument.
091         */
092        String argument();
093
094        /**
095         * Specify the body to be used.
096         * <p>
097         * This method must not return anything ; A {@code return this;} will be added
098         * by the framework at the end.
099         * 
100         * @return the body.
101         */
102        String[] body();
103}
