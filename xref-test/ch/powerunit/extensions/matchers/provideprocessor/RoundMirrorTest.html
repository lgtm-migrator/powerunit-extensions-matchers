72  	private Name mockName(String name) {
73  		Name mock = mock(Name.class);
74  		doReturn(name).when(mock).toString();
75  		return mock;
76  	}

78  	private void mockName(QualifiedNameable mock, String pkg, String name) {
79  		doReturn(mockName(name)).when(mock).getSimpleName();
80  		doReturn(mockName(pkg + "." + name)).when(mock).getQualifiedName();
81  	}

83  	private TypeElement mockTypeElement(String pkg, String name) {
84  		TypeElement mock = mock(TypeElement.class);
85  		doReturn(ElementKind.CLASS).when(mock).getKind();
86  		mockName(mock, pkg, name);
87  		return mock;
88  	}

90  	private void prepare() {
91  		when(processingEnv.getElementUtils()).thenReturn(elements);
92  		when(elements.getTypeElement(eq("java.lang.Object"))).thenReturn(te1);
93  		when(elements.getTypeElement(eq("ch.powerunit.extensions.matchers.ProvideMatchers"))).thenReturn(te1);
94  		when(elements.getTypeElement(eq("ch.powerunit.extensions.matchers.ProvideMatchers"))).thenReturn(te1);
95  		when(elements.getTypeElement(argThat(not(containsString("Matcher"))))).thenReturn(te1);
96  	}
