20  package ch.powerunit.extensions.matchers.provideprocessor.dsl;
21  
22  import ch.powerunit.Test;
23  import ch.powerunit.TestSuite;
24  
25  /**
26   * @author borettim
27   *
28   */
29  public class DSLMethodTest implements TestSuite {
30  
31  	@Test(fastFail = false)
32  	public void testDSLMethodTwoArgument() {
33  		DSLMethod m = DSLMethod.of("boolean isOK").addOneArgument("java.lang.String", "one")
34  				.addOneArgument("java.lang.String", "two").withImplementation("l1")
35  				.withJavadoc(new String[] { "l1", "l2" });
36  		assertThat(m.asStaticImplementation()).is(
37  				"/**\n * l1\n * l2\n */\npublic static boolean isOK(java.lang.String one,java.lang.String two) {\n  l1\n}\n");
38  		assertThat(m.asDefaultReference("target")).is(
39  				"/**\n * l1\n * l2\n */\ndefault boolean isOK(java.lang.String one,java.lang.String two) {\n  return target.isOK(one,two);\n}\n");
40  	}
41  
42  	@Test(fastFail = false)
43  	public void testDSLMethodOneArgument() {
44  		DSLMethod m = DSLMethod.of("boolean isOK").addOneArgument("java.lang.String", "one").withImplementation("l1")
45  				.withJavadoc(new String[] { "l1", "l2" });
46  		assertThat(m.asStaticImplementation())
47  				.is("/**\n * l1\n * l2\n */\npublic static boolean isOK(java.lang.String one) {\n  l1\n}\n");
48  		assertThat(m.asDefaultReference("target")).is(
49  				"/**\n * l1\n * l2\n */\ndefault boolean isOK(java.lang.String one) {\n  return target.isOK(one);\n}\n");
50  	}
51  
52  	@Test(fastFail = false)
53  	public void testDSLMethodZeroArgument() {
54  		DSLMethod m = DSLMethod.of("boolean isOK").withoutArgument().withImplementation("l1")
55  				.withJavadoc(new String[] { "l1", "l2" });
56  		assertThat(m.asStaticImplementation()).is("/**\n * l1\n * l2\n */\npublic static boolean isOK() {\n  l1\n}\n");
57  		assertThat(m.asDefaultReference("target"))
58  				.is("/**\n * l1\n * l2\n */\ndefault boolean isOK() {\n  return target.isOK();\n}\n");
59  	}
60  }
