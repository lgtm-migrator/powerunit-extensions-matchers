38  import ch.powerunit.extensions.matchers.provideprocessor.fields.CollectionFieldDescription;

42  /**
43   * @author borettim
44   *
45   */
46  public class JsonStringSpotifyAutomatedExtensionTest implements TestSuite {
47  
48  	@Mock
49  	private RoundMirror roundMirror;
50  
51  	@Mock
52  	private ProcessingEnvironment processingEnv;
53  
54  	@Mock
55  	private Elements elements;
56  
57  	@Mock
58  	private TypeElement targetTypeElement;
59  
60  	@Mock
61  	private TypeElement otherTypeElement;
62  
63  	@Mock
64  	private TypeMirror otherTypeMirror;
65  
66  	@Mock
67  	private StringFieldDescription stringField;
68  
69  	@Mock
70  	private CollectionFieldDescription collectionField;
71  
72  	private void prepare() {
73  		when(roundMirror.getProcessingEnv()).thenReturn(processingEnv);
74  		when(processingEnv.getElementUtils()).thenReturn(elements);
75  		when(elements.getTypeElement("com.spotify.hamcrest.jackson.JsonMatchers")).thenReturn(targetTypeElement);
76  		when(elements.getTypeElement("target")).thenReturn(otherTypeElement);
77  		when(otherTypeElement.asType()).thenReturn(otherTypeMirror);
78  
79  	}
80  
81  	@Rule
82  	public final TestRule rules = mockitoRule().around(before(this::prepare));
83  
84  	@Test
85  	public void testAcceptJsonMatcher() {
86  		JsonStringSpotifyAutomatedExtension tc = new JsonStringSpotifyAutomatedExtension(roundMirror);
87  		Collection<FieldDSLMethod> fdms = tc.acceptJsonMatcher(stringField);
88  		assertThatIterable(fdms).hasSize(1);
89  	}

91  }
