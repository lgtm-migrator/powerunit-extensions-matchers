65  	@Test(fastFail = false)
66  	public void testDefaultImplementationV1() {
67  		FieldDSLMethod fieldDSLMethod = FieldDSLMethodBuilder.of(fieldDescription).withDeclaration("a a")
68  				.withDefaultJavaDoc().havingImplementation("b");
69  		assertThat(fieldDSLMethod.asDSLMethod()).is(
70  				"/**\n * Add a validation on the field `fn`.\n * <p>\n *\n * <i>{@link fqnecof#fa This field is accessed by using this approach}.</i>\n * <p>\n * <b>In case method specifing a matcher on a fields are used several times, only the last setted matcher will be used.</b> \n * When several control must be done on a single field, hamcrest itself provides a way to combine several matchers.\n * (See for instance {@link org.hamcrest.Matchers#both(org.hamcrest.Matcher)}.\n *\n * @return the DSL to continue the construction of the matcher.\n */\n drm fn(a a);\n");
71  		assertThat(fieldDSLMethod.asImplementationMethod()).is("@Override\npublic  drm fn(a a) {\n  b\n}\n");
72  	}
