3   import ch.powerunit.Test;
4   import ch.powerunit.extensions.matchers.TestSuiteSupport;

6   public class ListJoiningTest implements TestSuiteSupport {
7   
8   	@Test(fastFail = false)
9   	public void testAsString() {
10  		assertThat(ListJoining.accepting(String.class).withMapper(s -> s + ";").withDelimiter("\n")
11  				.withPrefixAndSuffix("{", "}").asString("x", "y")).is("{x;\ny;}");
12  		assertThat(ListJoining.commaSeparated().asString("1", "2")).is("1,2");
13  		assertThat(ListJoining.commaSeparated(x -> x + ":").asString("1", "2")).is("1:,2:");
14  	}
15  }
