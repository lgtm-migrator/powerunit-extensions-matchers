43  	@Test(fastFail = false)
44  	public void testBoundsAsString() {
45  		TypeParameterElement tpe = mock(TypeParameterElement.class);
46  		TypeMirror type1 = mock(TypeMirror.class);
47  		TypeMirror type2 = mock(TypeMirror.class);
48  		doReturn(Collections.emptyList()).when(tpe).getBounds();
49  		assertThatFunction(this::boundsAsString, tpe).is("");
50  		when(type1.toString()).thenReturn("T1");
51  		doReturn(Collections.singletonList(type1)).when(tpe).getBounds();
52  		assertThatFunction(this::boundsAsString, tpe).is("T1");
53  		when(type1.toString()).thenReturn("T1");
54  		when(type2.toString()).thenReturn("T2");
55  		doReturn(Arrays.asList(type1, type2)).when(tpe).getBounds();
56  		assertThatFunction(this::boundsAsString, tpe).is("T1&T2");
57  	}

59  	@Test(fastFail = false)
60  	public void testGetGeneric() {
61  		TypeParameterElement tpe1 = mock(TypeParameterElement.class);
62  		TypeParameterElement tpe2 = mock(TypeParameterElement.class);
63  		TypeElement te = mock(TypeElement.class);
64  		doReturn(Collections.emptyList()).when(te).getTypeParameters();
65  		assertThatFunction(this::getGeneric, te).is("");
66  		doReturn(Collections.singletonList(tpe1)).when(te).getTypeParameters();
67  		when(tpe1.toString()).thenReturn("T1");
68  		assertThatFunction(this::getGeneric, te).is("<T1>");
69  		doReturn(Arrays.asList(tpe1, tpe2)).when(te).getTypeParameters();
70  		when(tpe1.toString()).thenReturn("T1");
71  		when(tpe2.toString()).thenReturn("T2");
72  		assertThatFunction(this::getGeneric, te).is("<T1,T2>");
73  	}

92  		when(tpe1.toString()).thenReturn("T1");
93  		when(tpe2.toString()).thenReturn("T2");

95  		when(type1.toString()).thenReturn("T1");
96  		when(type2.toString()).thenReturn("T2");
