20  package ch.powerunit.extensions.matchers.provideprocessor.fields;
21  
22  import java.util.ArrayList;
23  import java.util.Collection;
24  import java.util.Collections;
25  import java.util.List;
26  
27  import javax.lang.model.element.TypeElement;
28  
29  import ch.powerunit.extensions.matchers.common.ElementHelper;
30  import ch.powerunit.extensions.matchers.provideprocessor.Matchable;
31  import ch.powerunit.extensions.matchers.provideprocessor.ProvidesMatchersAnnotatedElementData;
32  
33  public class DefaultFieldDescription extends PrimitiveFieldDescription implements ElementHelper {
34  
35  	public DefaultFieldDescription(ProvidesMatchersAnnotatedElementData containingElementMirror,
36  			FieldDescriptionMirror mirror) {
37  		super(containingElementMirror, mirror);
38  	}
39  
40  	protected Collection<FieldDSLMethod> getSpecificFieldDslMethodFor() {
41  		return Collections.emptyList();
42  	}
43  
44  	@Override
45  	protected Collection<FieldDSLMethod> getFieldDslMethodFor() {
46  		List<FieldDSLMethod> tmp = new ArrayList<>(super.getFieldDslMethodFor());
47  		getTargetAsMatchable().filter(a -> mirror.getFieldTypeAsTypeElement().getTypeParameters().isEmpty())
48  				.map(this::createWithParent).ifPresent(tmp::add);
49  		tmp.add(FieldDSLMethodBuilder.of(this).withDeclaration("IsSameInstance", mirror.getFieldType() + " value")
50  				.withJavaDoc("",
51  						"value an expected value for the field, which will be compared that it is the same instance.",
52  						SEE_TEXT_FOR_IS_MATCHER)
53  				.havingDefault("(org.hamcrest.Matcher)" + MATCHERS + ".sameInstance((java.lang.Object)value)"));
54  		tmp.addAll(getSpecificFieldDslMethodFor());
55  		return tmp;
56  	}
57  
58  	private FieldDSLMethod createWithParent(Matchable target) {
59  		FieldDescriptionMirror fm = mirror;
60  		String fn = fm.getFieldName();
61  		TypeElement te = fm.getFieldTypeAsTypeElement();
62  		String name = getSimpleName(te);
63  		String targetMatcher = target.getFullyQualifiedNameOfGeneratedClass();
64  		return FieldDSLMethodBuilder.of(this).withExplicitDeclarationJavadocAndImplementation(
65  				targetMatcher + "." + name + "Matcher" + "<" + defaultReturnMethod + "> " + fn + "With()",
66  				"by starting a matcher for this field", targetMatcher + "." + name + "Matcher tmp = " + targetMatcher
67  						+ "." + target.getMethodNameDSLWithParent() + "(this);\n" + fn + "(tmp);\nreturn tmp;");
68  	}
69  
70  }
