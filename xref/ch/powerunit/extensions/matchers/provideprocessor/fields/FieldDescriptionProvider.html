20  package ch.powerunit.extensions.matchers.provideprocessor.fields;
21  
22  import javax.annotation.processing.ProcessingEnvironment;
23  import javax.lang.model.element.Element;
24  import javax.lang.model.element.ExecutableElement;
25  import javax.lang.model.type.ArrayType;
26  import javax.lang.model.type.DeclaredType;
27  import javax.lang.model.type.PrimitiveType;
28  import javax.lang.model.type.TypeMirror;
29  import javax.lang.model.type.TypeVariable;
30  import javax.lang.model.util.Elements;
31  import javax.lang.model.util.Types;
32  import javax.tools.Diagnostic.Kind;
33  
34  import ch.powerunit.extensions.matchers.IgnoreInMatcher;
35  import ch.powerunit.extensions.matchers.common.AbstractTypeKindVisitor;
36  import ch.powerunit.extensions.matchers.provideprocessor.ProvidesMatchersAnnotatedElementData;
37  import ch.powerunit.extensions.matchers.provideprocessor.RoundMirror;
38  
39  public final class FieldDescriptionProvider {
40  
41  	private FieldDescriptionProvider() {
42  	}
43  
44  	public static final class ExtractTypeVisitor
45  			extends AbstractTypeKindVisitor<AbstractFieldDescription, FieldDescriptionMirror, RoundMirror> {
46  
47  		private final ProvidesMatchersAnnotatedElementData containingElementMirror;
48  
49  		public ExtractTypeVisitor(ProvidesMatchersAnnotatedElementData containingElementMirror) {
50  			super(containingElementMirror.getRoundMirror());
51  			this.containingElementMirror = containingElementMirror;
52  		}
53  
54  		private boolean isAssignable(DeclaredType t, String target) {
55  			ProcessingEnvironment processingEnv = getProcessingEnv();
56  			Types types = processingEnv.getTypeUtils();
57  			Elements elements = processingEnv.getElementUtils();
58  			return types.isAssignable(t, types.erasure(elements.getTypeElement(target).asType()));
59  		}
60  
61  		@Override
62  		protected AbstractFieldDescription defaultAction(TypeMirror t, FieldDescriptionMirror mirror) {
63  			return new DefaultFieldDescription(containingElementMirror, mirror);
64  		}
65  
66  		@Override
67  		public AbstractFieldDescription visitPrimitive(PrimitiveType t, FieldDescriptionMirror mirror) {
68  			return new PrimitiveFieldDescription(containingElementMirror, mirror);
69  		}
70  
71  		@Override
72  		public AbstractFieldDescription visitArray(ArrayType t, FieldDescriptionMirror mirror) {
73  			return new ArrayFieldDescription(containingElementMirror, mirror);
74  		}
75  
76  		@Override
77  		public AbstractFieldDescription visitDeclared(DeclaredType t, FieldDescriptionMirror mirror) {
78  			if (isAssignable(t, "java.util.Optional")) {
79  				return new OptionalFieldDescription(containingElementMirror, mirror);
80  			} else if (isAssignable(t, "java.util.Map")) {
81  				return new MapFieldDescription(containingElementMirror, mirror);
82  			} else if (isAssignable(t, "java.util.Set") || isAssignable(t, "java.util.List")
83  					|| isAssignable(t, "java.util.Collection")) {
84  				return new CollectionFieldDescription(containingElementMirror, mirror);
85  			} else if (isAssignable(t, "java.lang.String")) {
86  				return new StringFieldDescription(containingElementMirror, mirror);
87  			} else if (isAssignable(t, "java.lang.Comparable")) {
88  				return new ComparableFieldDescription(containingElementMirror, mirror);
89  			} else if (isAssignable(t, "java.util.function.Supplier")) {
90  				return new SupplierFieldDescription(containingElementMirror, mirror);
91  			}
92  			return new DefaultFieldDescription(containingElementMirror, mirror);
93  		}
94  
95  		@Override
96  		public AbstractFieldDescription visitTypeVariable(TypeVariable t, FieldDescriptionMirror mirror) {
97  			return new DefaultFieldDescription(containingElementMirror, mirror);
98  		}
99  
100 		@Override
101 		public AbstractFieldDescription visitUnknown(TypeMirror t, FieldDescriptionMirror mirror) {
102 			getProcessingEnv().getMessager().printMessage(Kind.MANDATORY_WARNING, "Unsupported type element");
103 			return new DefaultFieldDescription(containingElementMirror, mirror);
104 		}
105 	}
106 
107 	public static boolean isIgnored(FieldDescriptionMirror mirror) {
108 		return mirror.getFieldElement().getAnnotation(IgnoreInMatcher.class) != null;
109 	}
110 
111 	public static AbstractFieldDescription of(ProvidesMatchersAnnotatedElementData containingElementMirror,
112 			FieldDescriptionMirror mirror) {
113 		if (isIgnored(mirror)) {
114 			return new IgnoreFieldDescription(containingElementMirror, mirror);
115 		}
116 		Element te = mirror.getFieldElement();
117 		return new ExtractTypeVisitor(containingElementMirror).visit(
118 				(te instanceof ExecutableElement) ? ((ExecutableElement) te).getReturnType() : te.asType(), mirror);
119 	}
120 
121 }
