20  package ch.powerunit.extensions.matchers.provideprocessor.fields;
21  
22  import static java.util.Optional.ofNullable;
23  
24  import java.util.ArrayList;
25  import java.util.Arrays;
26  import java.util.Collection;
27  import java.util.List;
28  
29  import ch.powerunit.extensions.matchers.provideprocessor.Matchable;
30  import ch.powerunit.extensions.matchers.provideprocessor.ProvidesMatchersAnnotatedElementData;
31  
32  public class CollectionFieldDescription extends DefaultFieldDescription {
33  
34  	private static final String DEFAULT_JAVADOC_MATCHER_ON_ELEMENTS = "matchersOnElements the matchers on the elements";
35  
36  	public CollectionFieldDescription(ProvidesMatchersAnnotatedElementData containingElementMirror,
37  			FieldDescriptionMirror mirror) {
38  		super(containingElementMirror, mirror);
39  	}
40  
41  	@Override
42  	protected Collection<FieldDSLMethod> getSpecificFieldDslMethodFor() {
43  		List<FieldDSLMethod> tmp = new ArrayList<>(getFieldDslMethodForIterableAndComparable());
44  		if (!"".equals(generic)) {
45  			tmp.addAll(getDslForIterableWithGeneric());
46  		}
47  		return tmp;
48  	}
49  
50  	public List<FieldDSLMethod> getFieldDslMethodForIterableAndComparable() {
51  		return Arrays.asList(
52  				getDslMethodBuilder().withSuffixDeclarationJavadocAndDefault("IsEmptyIterable",
53  						"that the iterable is empty", "(org.hamcrest.Matcher)" + MATCHERS + ".emptyIterable()"),
54  				getDslMethodBuilder().withSuffixDeclarationJavadocAndDefault("IsEmpty", "that the collection is empty",
55  						"(org.hamcrest.Matcher)" + MATCHERS + ".empty()"));
56  	}
57  
58  	public List<FieldDSLMethod> getDslForIterableWithGeneric() {
59  		String tgeneric = this.generic;
60  		String genericmatcher = "org.hamcrest.Matcher<" + tgeneric + ">";
61  		return Arrays.asList(
62  				getDslMethodBuilder().withDeclaration("Contains", tgeneric + "... elements")
63  						.withJavaDoc("that the iterable contains the received elements", "elements the elements",
64  								MATCHERS + "#contains(java.lang.Object[])")
65  						.havingDefault(MATCHERS + ".contains(elements)"),
66  				getDslMethodBuilder().withDeclaration("Contains", genericmatcher + "... matchersOnElements")
67  						.withJavaDoc("that the iterable contains the received elements, using matchers",
68  								DEFAULT_JAVADOC_MATCHER_ON_ELEMENTS, MATCHERS + "#contains(org.hamcrest.Matcher[])")
69  						.havingDefault(MATCHERS + ".contains(matchersOnElements)"),
70  				getDslMethodBuilder().withDeclaration("ContainsInAnyOrder", tgeneric + "... elements")
71  						.withJavaDoc("that the iterable contains the received elements in any order",
72  								"elements the elements", MATCHERS + "#containsInAnyOrder(java.lang.Object[])")
73  						.havingDefault(MATCHERS + ".containsInAnyOrder(elements)"),
74  				getDslMethodBuilder().withDeclaration("ContainsInAnyOrder", genericmatcher + "... matchersOnElements")
75  						.withJavaDoc("that the iterable contains the received elements, using matchers in any order",
76  								DEFAULT_JAVADOC_MATCHER_ON_ELEMENTS,
77  								MATCHERS + "#containsInAnyOrder(org.hamcrest.Matcher[])")
78  						.havingDefault(MATCHERS + ".containsInAnyOrder(matchersOnElements)"),
79  				getDslMethodBuilder()
80  						.withDeclaration("Contains",
81  								"java.util.List<org.hamcrest.Matcher<? super " + tgeneric + ">> matchersOnElements")
82  						.withJavaDoc("that the iterable contains the received elements, using list of matcher",
83  								DEFAULT_JAVADOC_MATCHER_ON_ELEMENTS, MATCHERS + "#contains(java.util.List)")
84  						.havingDefault(MATCHERS + ".contains(matchersOnElements)"));
85  	}
86  
87  	@Override
88  	public String getFieldCopy(String lhs, String rhs, String ignore) {
89  		if (!"".equals(generic)) {
90  			return getFieldCopyForList(lhs, rhs);
91  		}
92  		return super.getFieldCopy(lhs, rhs, ignore);
93  	}
94  
95  	protected String getFieldCopyForList(String lhs, String rhs) {
96  		String fieldAccessor = getFieldAccessor();
97  		String fieldName = getFieldName();
98  		return "if(" + rhs + "." + fieldAccessor + "==null) {\n  " + lhs + "." + fieldName + "(" + MATCHERS
99  				+ ".nullValue());\n} else if (" + rhs + "." + fieldAccessor + ".isEmpty()) {\n  " + lhs + "."
100 				+ fieldName + "IsEmptyIterable();\n} else {\n  " + lhs + "." + fieldName + "Contains(" + rhs + "."
101 				+ fieldAccessor + ".stream().map(" + generateMatcherBuilderReferenceFor(generic)
102 				+ ").collect(java.util.stream.Collectors.toList()));\n}";
103 	}
104 
105 	protected String generateMatcherBuilderReferenceFor(String generic) {
106 		return ofNullable(getByName(generic)).filter(Matchable::hasWithSameValue).map(t -> t.getWithSameValue(true))
107 				.orElse(MATCHERS + "::is");
108 	}
109 
110 }
