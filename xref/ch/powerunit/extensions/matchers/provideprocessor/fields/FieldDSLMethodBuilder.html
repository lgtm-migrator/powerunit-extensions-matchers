20  package ch.powerunit.extensions.matchers.provideprocessor.fields;
21  
22  import static java.util.Arrays.stream;
23  import static java.util.stream.Collectors.joining;
24  
25  import java.util.Optional;
26  
27  import ch.powerunit.extensions.matchers.provideprocessor.fields.lang.BuilderDeclaration;
28  import ch.powerunit.extensions.matchers.provideprocessor.fields.lang.BuilderImplementation;
29  import ch.powerunit.extensions.matchers.provideprocessor.fields.lang.BuilderJavadoc;
30  
31  public class FieldDSLMethodBuilder {
32  
33  	private static final String MATCHERS = "org.hamcrest.Matchers";
34  
35  	private static final String DEFAULT_JAVADOCFORMAT = "/**\n" + " * %1$s\n" + " * <p>\n" + " *\n" + " * <i>%2$s</i>\n"
36  			+ " * <p>\n"
37  			+ " * <b>In case method specifing a matcher on a fields are used several times, only the last setted matcher will be used.</b> \n"
38  			+ " * When several control must be done on a single field, hamcrest itself provides a way to combine several matchers.\n"
39  			+ " * (See for instance {@link " + MATCHERS + "#both(org.hamcrest.Matcher)}.\n" + " *\n" + "%3$s"
40  			+ " * @return the DSL to continue the construction of the matcher.\n" + "%4$s */";
41  
42  	public static BuilderDeclaration of(AbstractFieldDescription fieldDescription) {
43  		return new Builder(fieldDescription);
44  	}
45  
46  	public static class Builder implements BuilderDeclaration, BuilderJavadoc, BuilderImplementation {
47  		private final AbstractFieldDescription fieldDescription;
48  		private String declaration;
49  		private String javadoc;
50  
51  		private Builder(AbstractFieldDescription fieldDescription) {
52  			this.fieldDescription = fieldDescription;
53  		}
54  
55  		@Override
56  		public BuilderImplementation withJavaDoc(Optional<String> addToDescription, Optional<String> param,
57  				Optional<String> see) {
58  			javadoc = getJavaDocFor(fieldDescription, addToDescription, param, see);
59  			return this;
60  		}
61  
62  		@Override
63  		public FieldDSLMethod havingDefault(String innerMatcher) {
64  			return new FieldDSLMethod(buildDefaultDsl(fieldDescription, javadoc, declaration, innerMatcher), "");
65  		}
66  
67  		@Override
68  		public FieldDSLMethod havingImplementation(String body) {
69  			return new FieldDSLMethod(buildDsl(javadoc, declaration), buildImplementation(declaration, body));
70  		}
71  
72  		@Override
73  		public BuilderJavadoc withExplicitDeclaration(String declaration) {
74  			this.declaration = declaration;
75  			return this;
76  		}
77  
78  		@Override
79  		public BuilderJavadoc withGenericDeclaration(String generic, String postFix, String arguments) {
80  			this.declaration = generic + " " + String.format("%1$s %2$s%3$s(%4$s)",
81  					fieldDescription.getDefaultReturnMethod(), fieldDescription.getFieldName(), postFix, arguments);
82  			return this;
83  		}
84  
85  		@Override
86  		public FieldDSLMethod withExplicitDeclarationJavadocAndImplementation(String declaration,
87  				String addToDescription, String body) {
88  			return withExplicitDeclaration(declaration).withJavaDoc(addToDescription).havingImplementation(body);
89  		}
90  
91  		@Override
92  		public FieldDSLMethod withJavaDocAndDefault(String addToDescription, String innerMatcher) {
93  			return withJavaDoc(addToDescription).havingDefault(innerMatcher);
94  		}
95  
96  		@Override
97  		public FieldDSLMethod withSuffixDeclarationJavadocAndDefault(String declaration, String addToDescription,
98  				String innerMatcher) {
99  			return withSuffixDeclaration(declaration).withJavaDocAndDefault(addToDescription, innerMatcher);
100 		}
101 
102 	}
103 
104 	public static String getJavaDocFor(AbstractFieldDescription fieldDescription, Optional<String> addToDescription,
105 			Optional<String> param, Optional<String> see) {
106 		String linkToAccessor = String.format("{@link %1$s#%2$s This field is accessed by using this approach}.",
107 				fieldDescription.getFullyQualifiedNameEnclosingClassOfField(), fieldDescription.getFieldAccessor());
108 		String title = String.format("Add a validation on the field `%1$s`%2$s.", fieldDescription.getFieldName(),
109 				addToDescription.map(s -> " " + s).orElse(""));
110 		String paramString = param
111 				.map(t -> stream(t.split("\n")).map(l -> " * @param " + l + ".\n").collect(joining()))
112 				.orElse("");
113 		String seeString = see.map(s -> " * @see " + s + "\n").orElse("");
114 		return String.format(DEFAULT_JAVADOCFORMAT, title, linkToAccessor, paramString, seeString);
115 	}
116 
117 	public static String buildImplementation(String declaration, String body) {
118 		return String.format("@Override\npublic %1$s {\n  %2$s\n}\n", declaration, body.replaceAll("\\R", "\n" + "  "));
119 	}
120 
121 	public static String buildDsl(String javadoc, String declaration) {
122 		return String.format("%1$s\n%2$s;\n", javadoc.replaceAll("\\R", "\n"), declaration);
123 	}
124 
125 	public static String buildDefaultDsl(AbstractFieldDescription fieldDescription, String javadoc, String declaration,
126 			String innerMatcher) {
127 		return String.format("%1$s\ndefault %2$s{\n  return %3$s(%4$s);\n}", javadoc.replaceAll("\\R", "\n"),
128 				declaration, fieldDescription.getFieldName(), innerMatcher);
129 
130 	}
