20  package ch.powerunit.extensions.matchers.provideprocessor.fields;
21  
22  import static java.util.Optional.ofNullable;
23  
24  import java.util.ArrayList;
25  import java.util.Arrays;
26  import java.util.Collection;
27  
28  import ch.powerunit.extensions.matchers.provideprocessor.Matchable;
29  import ch.powerunit.extensions.matchers.provideprocessor.ProvidesMatchersAnnotatedElementData;
30  
31  public class OptionalFieldDescription extends DefaultFieldDescription {
32  
33  	public OptionalFieldDescription(ProvidesMatchersAnnotatedElementData containingElementMirror,
34  			FieldDescriptionMirror mirror) {
35  		super(containingElementMirror, mirror);
36  	}
37  
38  	@Override
39  	protected Collection<FieldDSLMethod> getSpecificFieldDslMethodFor() {
40  		Collection<FieldDSLMethod> dsl = new ArrayList<>();
41  		dsl.addAll(getPresentDSL());
42  		dsl.addAll(getAbsentDSL());
43  		return dsl;
44  	}
45  
46  	private Collection<FieldDSLMethod> getPresentDSL() {
47  		String fieldType = getFieldType();
48  		return Arrays.asList(
49  				getDslMethodBuilder().withSuffixDeclarationJavadocAndDefault("IsPresent", "with a present optional",
50  						"new org.hamcrest.CustomTypeSafeMatcher<" + fieldType
51  								+ ">(\"optional is present\"){ public boolean matchesSafely(" + fieldType
52  								+ " o) {return o.isPresent();}}"),
53  				getDslMethodBuilder().withDeclaration("IsPresentAndIs", generic + " value")
54  						.withJavaDoc("with a present optional having a specific value",
55  								"value the value the optional must have")
56  						.havingDefault("new org.hamcrest.TypeSafeMatcher<" + fieldType
57  								+ ">(){ public boolean matchesSafely(" + fieldType
58  								+ " o) {return o.isPresent() && o.get().equals(value);} public void describeTo(org.hamcrest.Description description) {description.appendText(\"optional is present and is \").appendValue(value);}}"),
59  				getDslMethodBuilder().withDeclaration("IsPresentAndIs", "org.hamcrest.Matcher<" + generic + "> matcher")
60  						.withJavaDoc("with a present optional matching a specified matcher",
61  								"matcher the matcher that must accept the optional value")
62  						.havingDefault("new org.hamcrest.TypeSafeMatcher<" + fieldType
63  								+ ">(){ public boolean matchesSafely(" + fieldType
64  								+ " o) {return o.isPresent() && matcher.matches(o.get());} public void describeTo(org.hamcrest.Description description) {description.appendText(\"optional is present and [\").appendDescriptionOf(matcher).appendText(\"]\");}}"));
65  	}
66  
67  	private Collection<FieldDSLMethod> getAbsentDSL() {
68  		String fieldType = getFieldType();
69  		return Arrays.asList(
70  				getDslMethodBuilder().withSuffixDeclarationJavadocAndDefault("IsNotPresent",
71  						"with a not present optional",
72  						"new org.hamcrest.CustomTypeSafeMatcher<" + fieldType
73  								+ ">(\"optional is not present\"){ public boolean matchesSafely(" + fieldType
74  								+ " o) {return !o.isPresent();}}"),
75  				getDslMethodBuilder().withSuffixDeclarationJavadocAndDefault("IsAbsent", "with an absent optional",
76  						"new org.hamcrest.CustomTypeSafeMatcher<" + fieldType
77  								+ ">(\"optional is not present\"){ public boolean matchesSafely(" + fieldType
78  								+ " o) {return !o.isPresent();}}"));
79  	}
80  
81  	@Override
82  	public String getFieldCopy(String lhs, String rhs, String ignore) {
83  		if (!"".equals(generic)) {
84  			return getFieldCopyForOptional(lhs, rhs, ignore);
85  		}
86  		return super.getFieldCopy(lhs, rhs, ignore);
87  	}
88  
89  	public String getFieldCopyForOptional(String lhs, String rhs, String paramForHasSameValue) {
90  		String fieldAccessor = getFieldAccessor();
91  		String fieldName = getFieldName();
92  		return "if(" + rhs + "." + fieldAccessor + "==null) {\n  " + lhs + "." + fieldName + "(" + MATCHERS
93  				+ ".nullValue());\n} else if (!" + rhs + "." + fieldAccessor + ".isPresent()) {\n  " + lhs + "."
94  				+ fieldName + "IsAbsent();\n} else {\n  " + lhs + "." + fieldName + "IsPresentAndIs("
95  				+ generateMatcherBuilderReferenceFor(generic, rhs + "." + fieldAccessor + ".get()",
96  						paramForHasSameValue)
97  				+ ");\n}";
98  	}
99  
100 	public String generateMatcherBuilderReferenceFor(String generic, String accessor, String paramForHasSameValue) {
101 		return ofNullable(getByName(generic)).filter(Matchable::hasWithSameValue).map(
102 				t -> t.getWithSameValue(false) + "(" + accessor + (t.supportIgnore() ? paramForHasSameValue : "") + ")")
103 				.orElse(MATCHERS + ".is(" + accessor + ")");
104 	}

106 }
