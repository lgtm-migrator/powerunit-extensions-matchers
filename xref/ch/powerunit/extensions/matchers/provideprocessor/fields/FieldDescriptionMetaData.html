20  package ch.powerunit.extensions.matchers.provideprocessor.fields;
21  
22  import static ch.powerunit.extensions.matchers.common.CommonUtils.toJavaSyntax;
23  import static java.lang.String.format;
24  import static java.util.stream.Collectors.joining;
25  
26  import java.util.Optional;
27  
28  import javax.lang.model.element.Element;
29  import javax.lang.model.type.DeclaredType;
30  import javax.lang.model.type.TypeMirror;
31  
32  import ch.powerunit.extensions.matchers.provideprocessor.Matchable;
33  import ch.powerunit.extensions.matchers.provideprocessor.ProvidesMatchersAnnotatedElementData;
34  import ch.powerunit.extensions.matchers.provideprocessor.helper.FeatureMatcher;
35  
36  public abstract class FieldDescriptionMetaData extends AbstractFieldDescriptionContainerMetaData {
37  
38  	public static final String SEE_TEXT_FOR_IS_MATCHER = "org.hamcrest.Matchers#is(java.lang.Object)";
39  	public static final String SEE_TEXT_FOR_HAMCREST_MATCHER = "org.hamcrest.Matchers The main class from hamcrest that provides default matchers.";
40  	public static final String MATCHERS = "org.hamcrest.Matchers";
41  
42  	protected final String generic;
43  	protected final String defaultReturnMethod;
44  	protected final FieldDescriptionMirror mirror;
45  
46  	public static final String computeGenericInformation(TypeMirror fieldTypeMirror) {
47  		if (fieldTypeMirror instanceof DeclaredType) {
48  			DeclaredType dt = ((DeclaredType) fieldTypeMirror);
49  			return dt.getTypeArguments().stream().map(Object::toString).collect(joining(","));
50  		}
51  		return "";
52  	}
53  
54  	public FieldDescriptionMetaData(ProvidesMatchersAnnotatedElementData containingElementMirror,
55  			FieldDescriptionMirror mirror) {
56  		super(containingElementMirror);
57  		this.mirror = mirror;
58  		TypeMirror fieldTypeMirror = mirror.getFieldTypeMirror();
59  		this.defaultReturnMethod = containingElementMirror.getDefaultReturnMethod();
60  		this.generic = computeGenericInformation(fieldTypeMirror);
61  	}
62  
63  	public String getMatcherForField() {
64  		return new FeatureMatcher(mirror.getMethodFieldName(), containingElementMirror.getFullGeneric(),
65  				getFullyQualifiedNameEnclosingClassOfField(), containingElementMirror.getGeneric(), getFieldType(),
66  				getFieldName(), "actual." + getFieldAccessor()).toString();
67  	}
68  
69  	public String getFieldCopy(String lhs, String rhs, String ignore) {
70  		return getTargetAsMatchable().filter(a -> mirror.getFieldTypeAsTypeElement().getTypeParameters().isEmpty())
71  				.filter(Matchable::hasWithSameValue)
72  				.map(p -> format(
73  						"%1$s.%6$s(%2$s.%3$s == null ? org.hamcrest.Matchers.nullValue() : %4$s(%2$s.%3$s%5$s));", lhs,
74  						rhs, getFieldAccessor(), p.getWithSameValue(false), p.supportIgnore() ? ignore : "",
75  						getFieldName()))
76  				.orElseGet(() -> format(
77  						"%1$s.%4$s((org.hamcrest.Matcher)org.hamcrest.Matchers.is((java.lang.Object)%2$s.%3$s));", lhs,
78  						rhs, getFieldAccessor(), getFieldName()));
79  	}
80  
81  	public String asMatchesSafely() {
82  		return format(
83  				"if(!%1$s.matches(actual)) {\n  mismatchDescription.appendText(\"[\"); %1$s.describeMismatch(actual,mismatchDescription); mismatchDescription.appendText(\"]\\n\");\n  result=false;\n}",
84  				getFieldName());
85  	}
86  
87  	public String asDescribeTo() {
88  		return "description.appendText(\"[\").appendDescriptionOf(" + getFieldName() + ").appendText(\"]\\n\");";
89  	}
90  
91  	public String asMatcherField() {
92  		return format("private %1$sMatcher %2$s = new %1$sMatcher(%3$s.anything(%4$s));", mirror.getMethodFieldName(),
93  				getFieldName(), MATCHERS, "");
94  	}
95  
96  	public String getFullyQualifiedNameEnclosingClassOfField() {
97  		return containingElementMirror.getFullyQualifiedNameOfClassAnnotatedWithProvideMatcher();
98  	}
99  
100 	public String getDefaultReturnMethod() {
101 		return defaultReturnMethod;
102 	}
103 
104 	public String getFieldAccessor() {
105 		return mirror.getFieldAccessor();
106 	}
107 
108 	public String getFieldName() {
109 		return mirror.getFieldName();
110 	}
111 
112 	public String getFieldType() {
113 		return mirror.getFieldType();
114 	}
115 
116 	public Element getFieldElement() {
117 		return mirror.getFieldElement();
118 	}
119 
120 	public FieldDescriptionMirror getMirror() {
121 		return mirror;
122 	}
123 
124 	public Optional<Matchable> getTargetAsMatchable() {
125 		return mirror.getMatchable(containingElementMirror.getRoundMirror());
126 	}
127 
128 	public String getGeneric() {
129 		return generic;
130 	}
131 
132 	public String generateMetadata(String className) {
133 		return "new " + className + "(" + toJavaSyntax(getFieldName()) + "," + toJavaSyntax(getFieldType()) + ","
134 				+ toJavaSyntax(getFieldAccessor()) + "," + toJavaSyntax(getClass().getSimpleName()) + ","
135 				+ Boolean.toString(this instanceof IgnoreFieldDescription) + ")";
136 	}
137 
138 }
