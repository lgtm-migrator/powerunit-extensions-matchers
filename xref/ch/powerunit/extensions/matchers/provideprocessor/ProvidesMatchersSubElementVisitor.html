20  package ch.powerunit.extensions.matchers.provideprocessor;
21  
22  import static ch.powerunit.extensions.matchers.common.CommonUtils.asStandardMethodName;
23  
24  import java.util.Optional;
25  import java.util.function.Function;
26  
27  import javax.lang.model.element.Element;
28  import javax.lang.model.element.ExecutableElement;
29  import javax.lang.model.element.VariableElement;
30  import javax.tools.Diagnostic.Kind;
31  
32  import ch.powerunit.extensions.matchers.common.AbstractSimpleElementVisitor;
33  import ch.powerunit.extensions.matchers.provideprocessor.fields.AbstractFieldDescription;
34  import ch.powerunit.extensions.matchers.provideprocessor.fields.FieldDescriptionMirror;
35  import ch.powerunit.extensions.matchers.provideprocessor.fields.FieldDescriptionProvider;
36  
37  /**
38   * @author borettim
39   *
40   */
41  public class ProvidesMatchersSubElementVisitor extends
42  		AbstractSimpleElementVisitor<Optional<AbstractFieldDescription>, ProvidesMatchersAnnotatedElementMatcherMirror, RoundMirror> {
43  
44  	private final Function<Element, Boolean> removeFromIgnoreList;
45  	private final NameExtractorVisitor extractNameVisitor;
46  
47  	public ProvidesMatchersSubElementVisitor(RoundMirror roundMirror) {
48  		super(roundMirror);
49  		this.removeFromIgnoreList = roundMirror::removeFromIgnoreList;
50  		this.extractNameVisitor = new NameExtractorVisitor(roundMirror);
51  	}
52  
53  	public Optional<AbstractFieldDescription> removeIfNeededAndThenReturn(
54  			Optional<AbstractFieldDescription> fieldDescription) {
55  		fieldDescription.map(AbstractFieldDescription::getFieldElement).ifPresent(removeFromIgnoreList::apply);
56  		return fieldDescription;
57  	}
58  
59  	@Override
60  	public Optional<AbstractFieldDescription> visitVariable(VariableElement e,
61  			ProvidesMatchersAnnotatedElementMatcherMirror p) {
62  		if (isPublic(e) && !isStatic(e)) {
63  			String fieldName = getSimpleName(e);
64  			return createFieldDescriptionIfApplicableAndRemoveElementFromListWhenApplicable(e, p, fieldName);
65  		}
66  		generateIfNeededErrorForNotSupportedElementAndRemoveIt("Check that this field is public and not static", e);
67  		return Optional.empty();
68  	}
69  
70  	@Override
71  	public Optional<AbstractFieldDescription> visitExecutable(ExecutableElement e,
72  			ProvidesMatchersAnnotatedElementMatcherMirror p) {
73  		if (isPublic(e) && e.getParameters().size() == 0 && !isStatic(e)) {
74  			String simpleName = getSimpleName(e);
75  			if (simpleName.matches("^((get)|(is)).*")) {
76  				return visiteExecutableGet(e, "^(get)|(is)", p);
77  			}
78  		}
79  		generateIfNeededErrorForNotSupportedElementAndRemoveIt(
80  				"Check that this method is public, doesn't have any parameter and is named isXXX or getXXX", e);
81  		return Optional.empty();
82  	}
83  
84  	private void generateIfNeededErrorForNotSupportedElementAndRemoveIt(String description, Element e) {
85  		if (removeFromIgnoreList.apply(e)) {
86  			getProcessingEnv().getMessager().printMessage(Kind.ERROR,
87  					"One of the annotation is not supported as this location ; " + description
88  							+ ". Since version 0.2.0 of powerunit-extension-matchers this is considered as an error.",
89  					e);
90  		}
91  	}
92  
93  	private Optional<AbstractFieldDescription> visiteExecutableGet(ExecutableElement e, String prefix,
94  			ProvidesMatchersAnnotatedElementMatcherMirror p) {
95  		return createFieldDescriptionIfApplicableAndRemoveElementFromListWhenApplicable(e, p,
96  				asStandardMethodName(getSimpleName(e).replaceFirst(prefix, "")));
97  	}
98  
99  	public Optional<AbstractFieldDescription> createFieldDescriptionIfApplicableAndRemoveElementFromListWhenApplicable(
100 			Element e, ProvidesMatchersAnnotatedElementMatcherMirror p, String fieldName) {
101 		return removeIfNeededAndThenReturn(
102 				((e instanceof ExecutableElement) ? ((ExecutableElement) e).getReturnType() : e.asType())
103 						.accept(extractNameVisitor, false).map(f -> FieldDescriptionProvider.of(() -> p,
104 								new FieldDescriptionMirror(() -> p, fieldName, f, e))));
105 	}
106 
107 	@Override
108 	protected Optional<AbstractFieldDescription> defaultAction(Element e,
109 			ProvidesMatchersAnnotatedElementMatcherMirror p) {
110 		return Optional.empty();
111 	}
112 
113 }
