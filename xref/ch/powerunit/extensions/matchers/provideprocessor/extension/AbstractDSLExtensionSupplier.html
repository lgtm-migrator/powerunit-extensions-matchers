20  package ch.powerunit.extensions.matchers.provideprocessor.extension;
21  
22  import static java.util.stream.Collectors.toList;
23  import static ch.powerunit.extensions.matchers.provideprocessor.dsl.DSLMethod.of;
24  import static java.util.stream.Collectors.joining;
25  
26  import java.util.Arrays;
27  import java.util.Collection;
28  import java.util.List;
29  import java.util.function.Supplier;
30  
31  import ch.powerunit.extensions.matchers.provideprocessor.dsl.DSLMethod;
32  
33  public abstract class AbstractDSLExtensionSupplier {
34  	protected final String targetName;
35  	protected final String returnType;
36  	protected final String methodName;
37  	protected final String targetMethodName;
38  
39  	public AbstractDSLExtensionSupplier(String targetName, String returnType, String methodName,
40  			String targetMethodName) {
41  		this.targetName = targetName;
42  		this.returnType = returnType;
43  		this.methodName = methodName;
44  		this.targetMethodName = targetMethodName;
45  	}
46  
47  	public abstract Collection<Supplier<DSLMethod>> asSuppliers();
48  
49  	public String[][] getSeveralParameter(boolean lastIsVarargs, String... name) {
50  		List<String[]> tmp = Arrays.stream(name).map(this::getOneParameter).collect(toList());
51  		if (lastIsVarargs) {
52  			tmp.get(name.length - 1)[0] += "...";
53  		}
54  		return tmp.toArray(new String[0][0]);
55  	}
56  
57  	public String[] getOneParameter(String name) {
58  		return new String[] { targetName, name };
59  	}
60  
61  	public String getSeveralWith(String... name) {
62  		return Arrays.stream(name).map(this::getOneWith).collect(joining(","));
63  	}
64  
65  	public String getOneWith(String name) {
66  		return targetMethodName + "(" + name + ")";
67  	}
68  
69  	public String[] generateSeveralWithImplementation(String matcher, String... parameters) {
70  		return new String[] { "java.util.List<org.hamcrest.Matcher<" + targetName
71  				+ ">> tmp = new java.util.ArrayList<>(java.util.Arrays.asList(" + getSeveralWith(parameters) + "));",
72  				"tmp.addAll(java.util.Arrays.stream(last).map(v->" + targetMethodName
73  						+ "(v)).collect(java.util.stream.Collectors.toList()));",
74  				"return " + matcher + "(tmp.toArray(new org.hamcrest.Matcher[0]));" };
75  	}
76  
77  	public DSLMethod generateSimpleDSLMethodFor(String javadoc[], String containerMatcher, String... parameters) {
78  		return of(returnType + " " + methodName).withArguments(getSeveralParameter(false, parameters))
79  				.withImplementation("return " + containerMatcher + "(" + getSeveralWith(parameters) + ");")
80  				.withJavadoc(javadoc);
81  	}
82  }
