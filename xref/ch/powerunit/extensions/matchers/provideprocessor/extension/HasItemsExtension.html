20  package ch.powerunit.extensions.matchers.provideprocessor.extension;
21  
22  import static ch.powerunit.extensions.matchers.provideprocessor.dsl.DSLMethod.of;
23  
24  import java.util.Arrays;
25  import java.util.Collection;
26  import java.util.function.Supplier;
27  
28  import ch.powerunit.extensions.matchers.ComplementaryExpositionMethod;
29  import ch.powerunit.extensions.matchers.provideprocessor.ProvidesMatchersAnnotatedElementData;
30  import ch.powerunit.extensions.matchers.provideprocessor.dsl.DSLMethod;
31  
32  public class HasItemsExtension implements DSLExtension {
33  
34  	public static final String CONTAINS_MATCHER = "org.hamcrest.Matchers.hasItems";
35  
36  	private static final String JAVADOC_DESCRIPTION = "Generate a hasItems matcher for this Object.";
37  
38  	@Override
39  	public ComplementaryExpositionMethod supportedEnum() {
40  		return ComplementaryExpositionMethod.HAS_ITEMS;
41  	}
42  
43  	@Override
44  	public Collection<Supplier<DSLMethod>> getDSLMethodFor(ProvidesMatchersAnnotatedElementData element) {
45  		return new HasItemSupplier(element).asSuppliers();
46  	}
47  
48  	public class HasItemSupplier extends AbstractDSLExtensionSupplier {
49  
50  		private final ProvidesMatchersAnnotatedElementData element;
51  
52  		public HasItemSupplier(ProvidesMatchersAnnotatedElementData element) {
53  			super(element.getFullyQualifiedNameOfClassAnnotatedWithProvideMatcherWithGeneric(),
54  					element.getFullGeneric() + " org.hamcrest.Matcher<java.lang.Iterable<"
55  							+ element.getFullyQualifiedNameOfClassAnnotatedWithProvideMatcherWithGeneric() + ">>",
56  					element.generateDSLMethodName("hasItems"), element.generateDSLWithSameValueMethodName());
57  			this.element = element;
58  		}
59  
60  		@Override
61  		public Collection<Supplier<DSLMethod>> asSuppliers() {
62  			return Arrays.asList(this::generateContainsN);
63  		}
64  
65  		public DSLMethod generateContainsN() {
66  			if (element.hasWithSameValue()) {
67  				return of(returnType + " " + methodName).withArguments(getSeveralParameter(true, "item"))
68  						.withImplementation("return " + CONTAINS_MATCHER + "(java.util.Arrays.stream(item).map(v->"
69  								+ targetMethodName
70  								+ "(v)).collect(java.util.stream.Collectors.toList()).toArray(new org.hamcrest.Matcher[0]));")
71  						.withJavadoc(JAVADOC_DESCRIPTION, "@param item the item to be matched", "@return the Matcher.");
72  			} else {
73  				element.printWarningMessage("Unable to apply the " + supportedEnum().name()
74  						+ " extension ; The target class doesn't support the WithSameValue() matcher");
75  				return null;
76  			}
77  		}
78  	}
