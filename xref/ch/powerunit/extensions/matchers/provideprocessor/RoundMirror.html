68  public class RoundMirror extends AbstractRoundMirrorReferenceToProcessingEnv {
69  
70  	private final Collection<AutomatedExtension> AUTOMATED_EXTENSIONS;
71  	private final Set<? extends Element> elementsWithPM;
72  	private final Map<Class<?>, Set<? extends Element>> elementsWithOtherAnnotations;
73  	private final Map<String, ProvidesMatchersAnnotatedElementMirror> alias = new HashMap<>();
74  
75  	public RoundMirror(RoundEnvironment roundEnv, ProcessingEnvironment processingEnv) {
76  		super(roundEnv, processingEnv);
77  		this.elementsWithOtherAnnotations = new HashMap<>();
78  		this.elementsWithPM = roundEnv.getElementsAnnotatedWith(ProvideMatchers.class);
79  		elementsWithOtherAnnotations.put(IgnoreInMatcher.class,
80  				new HashSet<>(roundEnv.getElementsAnnotatedWith(IgnoreInMatcher.class)));
81  		elementsWithOtherAnnotations.put(AddToMatcher.class,
82  				new HashSet<>(roundEnv.getElementsAnnotatedWith(AddToMatcher.class)));
83  		elementsWithOtherAnnotations.put(AddToMatchers.class,
84  				new HashSet<>(roundEnv.getElementsAnnotatedWith(AddToMatchers.class)));
85  		AUTOMATED_EXTENSIONS = getDefaultExtension().stream().filter(AutomatedExtension::isPresent)
86  				.collect(collectingAndThen(toList(), Collections::unmodifiableList));
87  	}
88  
89  	private final List<AutomatedExtension> getDefaultExtension() {
90  		return Arrays.asList(new LocalDateMatchersAutomatedExtension(this),
91  				new LocalDateTimeMatchersAutomatedExtension(this), new LocalTimeMatchersAutomatedExtension(this),
92  				new ZonedDateTimeMatchersAutomatedExtension(this),
93  				new CollectionHamcrestUtilityAutomatedExtension(this), new JsonStringSpotifyAutomatedExtension(this),
94  				new DefaultBeanMatchersAutomatedExtension(this), new DefaultJsonNodeJacksonAutomatedExtension(this));
95  	}
96  
97  	public Collection<ProvidesMatchersAnnotatedElementMirror> parse() {
98  		ProvidesMatchersElementVisitor providesMatchersElementVisitor = new ProvidesMatchersElementVisitor(this);
99  		elementsWithPM.stream().filter(e -> roundEnv.getRootElements().contains(e))
100 				.map(e -> e.accept(providesMatchersElementVisitor, null)).filter(Optional::isPresent)
101 				.map(t -> new ProvidesMatchersAnnotatedElementMirror(t.get(), this))
102 				.forEach(a -> alias.put(a.getFullyQualifiedNameOfClassAnnotated(), a));
103 
104 		doErrorforAllElements();
105 		return alias.values();
106 	}
107 
108 	private void doErrorforAllElements() {
109 		elementsWithOtherAnnotations.entrySet().stream().forEach(e -> doErrorForElement(e.getValue(), e.getKey()));
110 	}
111 
112 	private void doErrorForElement(Set<? extends Element> elements, Class<?> aa) {
113 		elements.stream().forEach(e -> getMessager().printMessage(Kind.ERROR, "Annotation @" + aa.getName()
114 				+ " not supported at this location ; The surrounding class is not annotated with @ProvideMatchers. Since version 0.2.0 of powerunit-extension-matchers this is considered as an error.",
115 				e, findAnnotationMirrorFor(e, aa)));
116 	}
117 
118 	private AnnotationMirror findAnnotationMirrorFor(Element e, Class<?> aa) {
119 		String aaName = aa.getName().toString();
120 		return e.getAnnotationMirrors().stream()
121 				.filter(a -> a.getAnnotationType().equals(getElementUtils().getTypeElement(aaName).asType())).findAny()
122 				.orElse(null);
123 	}
124 
125 	public boolean removeFromIgnoreList(Element e) {
126 		return elementsWithOtherAnnotations.values().stream().map(t -> t.remove(e)).filter(t -> t).findAny()
127 				.orElse(false);
128 	}
129 
130 	public Matchable getByName(String name) {
131 		return Optional.ofNullable((Matchable) alias.get(name)).orElseGet(() -> Optional
132 				.ofNullable(getElementUtils().getTypeElement(name)).map(this::lookupMatchableByType).orElse(null));
133 
134 	}
135 
136 	public Matchable lookupMatchableByType(TypeElement type) {
137 		return Optional.ofNullable(getElementUtils().getTypeElement(getQualifiedName(type) + "Matchers"))
138 				.map(m -> lookupMatchableByTypeAndMatchers(type, m)).orElse(null);
139 	}
140 
141 	private Matchable lookupMatchableByTypeAndMatchers(TypeElement type, TypeElement guestMatcher) {
142 		List<? extends Element> guestMatcherEnclosed = guestMatcher.getEnclosedElements();
143 		List<TypeElement> types = typesIn(guestMatcherEnclosed);
144 		Optional<Long> compatibilityField = types.stream().filter(t -> isSimpleName(t, "Metadata"))
145 				.flatMap(t -> fieldsIn(t.getEnclosedElements()).stream()).filter(t -> isSimpleName(t, "COMPATIBILITY"))
146 				.map(VariableElement::getConstantValue).filter(Objects::nonNull).filter(Long.class::isInstance)
147 				.map(Long.class::cast).findAny();
148 		if (!compatibilityField.isPresent()) {
149 			return null;// NOT FOUND
150 		}
151 		// In future, verify compatiblity
152 		String guestMatcherName = getSimpleName(type) + "Matcher";
153 		if (!types.stream().anyMatch(t -> isSimpleName(t, guestMatcherName))) {
154 			return null;// NOT FOUND
155 		}
156 		String shortMethodClassName = asStandardMethodName(getSimpleName(type));
157 		String withSameValue = shortMethodClassName + "WithSameValue";
158 		boolean hasSameValue = methodsIn(guestMatcherEnclosed).stream().filter(t -> isSimpleName(t, withSameValue))
159 				.anyMatch(this::isStatic);
160 		return Matchable.of(getQualifiedName(guestMatcher), shortMethodClassName, guestMatcherName, hasSameValue,
161 				compatibilityField.get());
162 	}
163 
164 	public AnnotationMirror getProvideMatchersAnnotation(Element e) {
165 		return getElementUtils().getAllAnnotationMirrors(e).stream()
166 				.filter(a -> a.getAnnotationType().equals(provideMatchersMirror)).findAny().orElse(null);
167 	}

169 	public Collection<Supplier<DSLMethod>> getDSLMethodFor(ProvidesMatchersAnnotatedElementData target) {
170 		return AUTOMATED_EXTENSIONS.stream().map(ae -> ae.accept(target)).flatMap(Collection::stream).collect(toList());
171 	}
172 
173 	public Collection<FieldDSLMethod> getFieldDSLMethodFor(AbstractFieldDescription target) {
174 		return AUTOMATED_EXTENSIONS.stream().map(ae -> ae.accept(target)).flatMap(Collection::stream).collect(toList());
175 	}
