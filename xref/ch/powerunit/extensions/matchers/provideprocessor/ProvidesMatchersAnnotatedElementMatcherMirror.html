96  	private List<AbstractFieldDescription> generateFields(TypeElement typeElement,
97  			ProvidesMatchersSubElementVisitor providesMatchersSubElementVisitor) {
98  		return typeElement.getEnclosedElements().stream().map(ie -> ie.accept(providesMatchersSubElementVisitor, this))
99  				.filter(Optional::isPresent).map(Optional::get)
100 				.collect(collectingAndThen(groupingBy(FieldDescriptionMetaData::getFieldName, reducing(
101 						ProvidesMatchersAnnotatedElementMatcherMirror::reduceByOrderingOnIgnoreFieldDescription)),
102 						Map::values))
103 				.stream().filter(Optional::isPresent).map(Optional::get).sorted(COMPARING_FIELD_BY_NAME)
104 				.collect(toList());
105 	}

107 	public ProvidesMatchersAnnotatedElementMatcherMirror(TypeElement typeElement, RoundMirror roundMirror) {
108 		super(typeElement, roundMirror);
109 		this.fields = unmodifiableList(generateFields(typeElement, new ProvidesMatchersSubElementVisitor(roundMirror)));
110 		this.fieldsMatcher = JOIN_FIELD_MATCHER.asString(fields) + "\n";
111 	}

117 						getFullyQualifiedNameOfClassAnnotatedWithProvideMatcherWithGeneric(), getFullGenericParent(),
118 						generic, getGenericParent(), JOIN_FIELD_DSL_INTERFACE.asString(fields)));
119 	}

121 	protected String generatePrivateImplementation() {
122 		return addPrefix("  ", format(hasSuperClass() ? IMPLEMENTATION_PARENT_FORMAT : IMPLEMENTATION_NO_PARENT_FORMAT,
123 				getSimpleNameOfGeneratedImplementationMatcher(), getFullGenericParent(),
124 				getFullyQualifiedNameOfClassAnnotatedWithProvideMatcherWithGeneric(),
125 				getSimpleNameOfGeneratedInterfaceMatcherWithGenericParent(), getFullyQualifiedNameOfClassAnnotated(),
126 				JOIN_FIELD_MATCHERFIELD.asString(fields), JOIN_FIELD_IMPL.asString(fields),
127 				JOIN_FIELD_MATCHES_SAFELY.asString(fields), JOIN_FIELD_DESCRIBE_TO.asString(fields),
128 				fullyQualifiedNameOfSuperClassOfClassAnnotated.orElse("")));

130 	}

132 	public String generateMatchers() {
133 		return new StringBuilder(DEFAULT_FEATUREMATCHER_FORCONVERTER).append(fieldsMatcher)
134 				.append(fullyQualifiedNameOfSuperClassOfClassAnnotated.map(this::generateParentMatcher).orElse(""))
135 				.toString();
136 	}

141 	}
