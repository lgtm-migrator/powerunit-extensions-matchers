54  	private final Collection<Supplier<Collection<DSLMethod>>> dslProvider;
55  
56  	private static final String MATCHER_FORMAT = RessourceLoaderHelper
57  			.loadRessource(ProvidesMatchersAnnotatedElementMirror.class, "Matchers.txt");
58  
59  	private static final String POSTPROCESSOR_FORMAT = addPrefix("  ",
60  			RessourceLoaderHelper.loadRessource(ProvidesMatchersAnnotatedElementMirror.class, "PostProcessor.txt"));
61  
62  	private static final ListJoining<DSLMethod> JOIN_DSL_METHOD = ListJoining
63  			.joinWithMapperAndDelimiter(m -> addPrefix("  ", m.asStaticImplementation()), "\n");
64  
65  	private static Supplier<Collection<DSLMethod>> asCollection(Supplier<DSLMethod> input) {
66  		return () -> singletonList(input.get());
67  	}

69  	public ProvidesMatchersAnnotatedElementMirror(TypeElement typeElement, RoundMirror roundMirror) {
70  		super(typeElement, roundMirror);
71  		List<Supplier<Collection<DSLMethod>>> tmp = new ArrayList<>(Arrays.asList(
72  				asCollection(this::generateDefaultDSLStarter), asCollection(this::generateDefaultForChainingDSLStarter),
73  				asCollection(this::generateMatcherClassMethod)));
74  		if (hasSuperClass()) {
75  			tmp.add(asCollection(this::generateParentDSLStarter));
76  			tmp.add(() -> ProvidesMatchersWithSameValueHelper.generateParentValueDSLStarter(this));
77  			if (((TypeElement) roundMirror.getTypeUtils().asElement(element.getSuperclass())).getTypeParameters()
78  					.isEmpty()) {
79  				tmp.add(asCollection(this::generateParentInSameRoundWithChaningDSLStarter));
80  			}
81  		} else {
82  			tmp.add(() -> ProvidesMatchersWithSameValueHelper.generateNoParentValueDSLStarter(this));
83  		}
84  		tmp.addAll(ofNullable(getDSLExtension()).orElseGet(Collections::emptyList).stream()
85  				.map(t -> t.getDSLMethodFor(() -> this)).flatMap(Collection::stream)
86  				.map(ProvidesMatchersAnnotatedElementMirror::asCollection).collect(toList()));
87  		tmp.addAll(roundMirror.getDSLMethodFor(() -> this).stream()
88  				.map(ProvidesMatchersAnnotatedElementMirror::asCollection).collect(toList()));
89  		this.dslProvider = unmodifiableList(tmp);
90  	}
91  
92  	public Collection<DSLMethod> process() {
93  		Element te = element;
94  		return processFileWithIOExceptionAndResult(
95  				() -> getFiler().createSourceFile(getFullyQualifiedNameOfGeneratedClass(), te),
96  				jfo -> new Formatter(new PrintWriter(jfo.openWriter())), output -> {
97  					Collection<DSLMethod> tmp = generateDSLStarter();
98  					output.format(MATCHER_FORMAT, getPackageNameOfGeneratedClass(),
99  							getFullyQualifiedNameOfClassAnnotated(),
100 							generateGeneratedAnnotation(ProvidesMatchersAnnotationsProcessor.class, comments()),
101 							getSimpleNameOfGeneratedClass(), generateMatchers(), generatePublicInterface(),
102 							generatePrivateImplementation(), JOIN_DSL_METHOD.asString(tmp), POSTPROCESSOR_FORMAT,
103 							generateMetadata());
104 					output.flush();
105 					return tmp;
106 				}, e -> traceErrorAndDump(this, e, te));
107 	}

109 	public Collection<DSLMethod> generateDSLStarter() {
110 		return dslProvider.stream().map(Supplier::get).filter(Objects::nonNull).flatMap(Collection::stream)
111 				.filter(Objects::nonNull).collect(toList());
112 	}

114 	public String getDefaultStarterBody(boolean withParentBuilder) {
115 		String targetImpl = withParentBuilder ? getSimpleNameOfGeneratedImplementationMatcherWithGenericParent()
116 				: getSimpleNameOfGeneratedImplementationMatcherWithGenericNoParent();
117 		boolean withSuper = hasSuperClass();
118 		return "return new " + targetImpl + "(" + (withSuper ? "org.hamcrest.Matchers.anything()" : "")
119 				+ (withSuper && withParentBuilder ? "," : "") + (withParentBuilder ? "parentBuilder" : "") + ");";
120 	}

125 	}

132 	public DSLMethod generateDefaultDSLStarter() {
133 		return of(fullGeneric + " " + getFullyQualifiedNameOfGeneratedClass() + "."
134 				+ getSimpleNameOfGeneratedInterfaceMatcherWithGenericNoParent() + " " + methodShortClassName + "With")
135 						.withImplementation(getDefaultStarterBody(false))
136 						.withJavadoc(generateDefaultJavaDocWithDSLStarter(empty(), "the DSL matcher", false));
137 	}
138 
139 	public String getJavadocForDSLStarter() {
140 		return "The returned builder (which is also a Matcher), at this point accepts any object that is a "
141 				+ getDefaultLinkForAnnotatedClass() + ".";
142 	}
143 
144 	public DSLMethod generateDefaultForChainingDSLStarter() {
145 		return of(getFullGenericParent() + " " + getFullyQualifiedNameOfGeneratedClass() + "."
146 				+ getSimpleNameOfGeneratedInterfaceMatcherWithGenericParent() + " " + getMethodNameDSLWithParent())
147 						.withOneArgument("_PARENT", "parentBuilder").withImplementation(getDefaultStarterBody(true))
148 						.withJavadoc(generateDefaultJavaDocWithDSLStarter(
149 								Optional.of("parentBuilder the parentBuilder."), "the DSL matcher", true));
150 	}
151 
152 	public DSLMethod generateParentDSLStarter() {
153 		String mscn = methodShortClassName;
154 		String fqngc = getFullyQualifiedNameOfGeneratedClass();
155 		return of(fullGeneric + " " + fqngc + "." + getSimpleNameOfGeneratedInterfaceMatcherWithGenericNoParent() + " "
156 				+ mscn + "With").withOneArgument(
157 						"org.hamcrest.Matcher<? super " + fullyQualifiedNameOfSuperClassOfClassAnnotated.get() + ">",
158 						"matcherOnParent")
159 						.withImplementation(
160 								"return new " + getSimpleNameOfGeneratedImplementationMatcherWithGenericNoParent()
161 										+ "(matcherOnParent);")
162 						.withJavadoc(generateDefaultJavaDocWithoutDSLStarter(
163 								Optional.of("matcherOnParent the matcher on the parent data."), "the DSL matcher",
164 								false));
165 	}
166 
167 	public DSLMethod generateParentInSameRoundWithChaningDSLStarter() {
168 		String implGenericNoParent = getSimpleNameOfGeneratedImplementationMatcherWithGenericNoParent();
169 		return getParentMirror().map(parentMirror -> {
170 			String pmfqngc = parentMirror.getFullyQualifiedNameOfGeneratedClass();
171 			String parentSimpleName = parentMirror.getSimpleNameOfGeneratedInterfaceMatcher();
172 			return of(fullGeneric + " " + pmfqngc + "." + parentSimpleName + genericForChaining + " "
173 					+ getMethodNameDSLWithParent())
174 							.withImplementation(
175 									implGenericNoParent + " m=new " + implGenericNoParent
176 											+ "(org.hamcrest.Matchers.anything());",
177 									pmfqngc + "." + parentSimpleName + " tmp = " + pmfqngc + "."
178 											+ parentMirror.getMethodNameDSLWithParent() + "(m);",
179 									"m._parent = new SuperClassMatcher(tmp);", "return tmp;")
180 							.withJavadoc(generateDefaultJavaDoc(empty(), empty(), "the DSL matcher", false));
181 		}).orElse(null);
182 	}

184 	public DSLMethod generateMatcherClassMethod() {
185 		return of(getFullGenericParent()+" Class<" + getFullyQualifiedNameOfGeneratedClass() + "."
186 				+ getSimpleNameOfGeneratedInterfaceMatcherWithGenericParent() + "> "
187 				+ getMethodShortClassName() + "MatcherClass").withoutArgument()
188 						.withImplementation("return (Class)" + getSimpleNameOfGeneratedInterfaceMatcher() + ".class;")
189 						.withJavadoc(
190 								"/**\n * Helper method to retrieve the Class of the matcher interface.\n * @return the class.\n */\n");
191 	}
