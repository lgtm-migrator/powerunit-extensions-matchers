20  package ch.powerunit.extensions.matchers.provideprocessor.dsl;
21  
22  import static ch.powerunit.extensions.matchers.common.ListJoining.joinWithMapper;
23  
24  import java.util.ArrayList;
25  import java.util.List;
26  import java.util.Objects;
27  import java.util.regex.Matcher;
28  import java.util.regex.Pattern;
29  
30  import ch.powerunit.extensions.matchers.common.ListJoining;
31  import ch.powerunit.extensions.matchers.provideprocessor.dsl.lang.DSLMethodArgument;
32  import ch.powerunit.extensions.matchers.provideprocessor.dsl.lang.DSLMethodImplementation;
33  import ch.powerunit.extensions.matchers.provideprocessor.dsl.lang.DSLMethodJavadoc;
34  
35  /**
36   * @author borettim
37   * 
38   */
39  public final class DSLMethod {
40  	public static final Pattern DECLARATION_PARSER = Pattern.compile("^\\s*.*\\s+([0-9A-Za-z_]+)\\s*$");
41  
42  	private static final ListJoining<String[]> ARGUMENTS_JOIN = joinWithMapper((String a[]) -> a[0] + " " + a[1])
43  			.withCommaDelimiter().withPrefixAndSuffix("(", ")");
44  
45  	private static final ListJoining<String[]> ARGUMENTNAMES_JOIN = joinWithMapper((String a[]) -> a[1])
46  			.withCommaDelimiter().withPrefixAndSuffix("(", ")");
47  
48  	private static final ListJoining<String> IMPLEMENTATION_JOIN = joinWithMapper((String s) -> "  " + s)
49  			.withDelimiter("\n").withPrefixAndSuffix("", "\n");
50  
51  	private final String implementation;
52  	private final String defaultReference;
53  
54  	private static class Builder implements DSLMethodArgument, DSLMethodImplementation, DSLMethodJavadoc {
55  
56  		private final String declaration;
57  		private final List<String[]> arguments = new ArrayList<>();
58  		private String implementation[];
59  
60  		public Builder(String declaration) {
61  			this.declaration = declaration;
62  		}
63  
64  		@Override
65  		public DSLMethod withJavadoc(String javadoc) {
66  			return new DSLMethod(javadoc, declaration, arguments.toArray(new String[][] {}), implementation);
67  		}
68  
69  		@Override
70  		public DSLMethodJavadoc withImplementation(String... implementation) {
71  			this.implementation = implementation;
72  			return this;
73  		}
74  
75  		@Override
76  		public DSLMethodArgument addOneArgument(String type, String name) {
77  			arguments.add(new String[] { Objects.requireNonNull(type, "type can't be null"),
78  					Objects.requireNonNull(name, "name can't be null") });
79  			return this;
80  		}
81  
82  	}
83  
84  	public static DSLMethodArgument of(String declaration) {
85  		return new Builder(declaration);
86  	}
87  
88  	public DSLMethod(String javadoc, String declaration, String arguments[][], String implementation[]) {
89  		Matcher m = DECLARATION_PARSER.matcher(declaration);
90  		if (!m.matches()) {
91  			throw new IllegalArgumentException("Unable to parse the received declaration");
92  		}
93  		String implementations = IMPLEMENTATION_JOIN.asString(implementation);
94  		String fullDeclaration = declaration + ARGUMENTS_JOIN.asString(arguments);
95  		String fullMethodName = m.group(1) + ARGUMENTNAMES_JOIN.asString(arguments);
96  		this.implementation = javadoc + "public static " + fullDeclaration + " {\n" + implementations + "}\n";
97  		this.defaultReference = javadoc + "default " + fullDeclaration + " {\n  return %1$s." + fullMethodName
98  				+ ";\n}\n";
99  	}
100 
101 	public String asStaticImplementation() {
102 		return implementation;
103 	}
104 
105 	public String asDefaultReference(String target) {
106 		return String.format(defaultReference, target);
107 	}
108 
109 }
