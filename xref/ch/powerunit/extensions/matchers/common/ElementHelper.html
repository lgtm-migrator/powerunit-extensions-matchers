20  package ch.powerunit.extensions.matchers.common;
21  
22  import static ch.powerunit.extensions.matchers.common.ListJoining.accepting;
23  
24  import java.util.List;
25  
26  import javax.lang.model.element.Element;
27  import javax.lang.model.element.Modifier;
28  import javax.lang.model.element.Parameterizable;
29  import javax.lang.model.element.QualifiedNameable;
30  import javax.lang.model.element.TypeParameterElement;
31  import javax.lang.model.type.TypeMirror;
32  
33  /**
34   * @author borettim
35   *
36   */
37  public interface ElementHelper {
38  
39  	static final ListJoining<TypeMirror> TYPE_PARAMETER_BOUND_AS_LIST = accepting(TypeMirror.class).withToStringMapper()
40  			.withDelimiter("&").withoutSuffixAndPrefix();
41  
42  	static final ListJoining<TypeMirror> TYPE_PARAMETER_BOUND_AS_LIST_WITH_EXTENDS = accepting(TypeMirror.class)
43  			.withToStringMapper().withDelimiter("&").withOptionalPrefixAndSuffix(" extends ", "");
44  
45  	static final ListJoining<TypeParameterElement> TYPE_PARAMETER_SIMPLE_AS_LIST = accepting(TypeParameterElement.class)
46  			.withToStringMapper().withCommaDelimiter().withOptionalPrefixAndSuffix("<", ">");
47  
48  	@SuppressWarnings("unchecked")
49  	static final ListJoining<TypeParameterElement> TYPE_PARAMETER_FULL_AS_LIST = accepting(TypeParameterElement.class)
50  			.withMapper(t -> t.toString()
51  					+ TYPE_PARAMETER_BOUND_AS_LIST_WITH_EXTENDS.asString((List<TypeMirror>) t.getBounds()))
52  			.withCommaDelimiter().withOptionalPrefixAndSuffix("<", ">");
53  
54  	default String getSimpleName(Element e) {
55  		return e.getSimpleName().toString();
56  	}
57  
58  	default boolean isSimpleName(Element e, String name) {
59  		return name.equals(getSimpleName(e));
60  	}
61  
62  	default String getQualifiedName(QualifiedNameable e) {
63  		return e.getQualifiedName().toString();
64  	}
65  
66  	default boolean isStatic(Element e) {
67  		return e.getModifiers().contains(Modifier.STATIC);
68  	}
69  
70  	default boolean isPublic(Element e) {
71  		return e.getModifiers().contains(Modifier.PUBLIC);
72  	}
73  
74  	@SuppressWarnings("unchecked")
75  	default String boundsAsString(TypeParameterElement e) {
76  		return TYPE_PARAMETER_BOUND_AS_LIST.asString((List<TypeMirror>) e.getBounds());
77  	}
78  
79  	@SuppressWarnings("unchecked")
80  	default String getGeneric(Parameterizable typeElement) {
81  		return TYPE_PARAMETER_SIMPLE_AS_LIST.asString((List<TypeParameterElement>) typeElement.getTypeParameters());
82  	}
83  
84  	@SuppressWarnings("unchecked")
85  	default String getFullGeneric(Parameterizable typeElement) {
86  		return TYPE_PARAMETER_FULL_AS_LIST.asString((List<TypeParameterElement>) typeElement.getTypeParameters());
87  	}
88  }
